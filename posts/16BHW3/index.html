1. get_message_db():

This function creates the database of messages messages_db, a table messages, and return the connection g.message.db

# Database connection
def get_message_db():
    try:
        return g.message_db
    except AttributeError:
        # Connect to the database message_db, ensuring the connection is an attribute of g.
        g.message_db = sqlite3.connect("messages_db.sqlite")
        # This SQL Command is to first check whether a table called messages exists in the created database
        # And then it creates a table which includes two text columns (handle and message)
        cmd = '''
        CREATE TABLE IF NOT EXISTS messages (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            handle TEXT NOT NULL,
            message TEXT NOT NULL
        )
        '''
        cursor = g.message_db.cursor()
        cursor.execute(cmd)
        g.message_db.commit()
        # Return the connection g.message_db
        return g.message_db

2. insert_message(request):

This function handles inserting a user messgae into the dabase of messages

# Insert values into database
def insert_message(request):
    #This SQL command insert both handle and message into two corresponding columns in the table
    cmd = 'INSERT INTO messages (handle, message) VALUES (?, ?)'
    #Here I specify the column name of handle as "name"
    handle = request.form["name"]
    message = request.form["message"]
    db = get_message_db()
    #Using cursor to insert the message into message database
    cursor = db.cursor()
    #We need provide the handle, the message itself.
    cursor.execute(cmd, (handle, message))
    #When working directly with SQL commands, it's neccessary to run db.commit() after inserting a row into db
    #This is to ensure row insertion has been saved
    g.message_db.commit()

3. submit (): render_template() the submit.html template

This function ensure transmit and receive data, supporting both POST and GET methods

# Submit message route
# 'POST' and 'GET' splites two returned pages. Ensure the page both transmit and receive data.
@app.route('/submit', methods=['POST', 'GET'])

def submit():
    # In POST case, call insert_message () and if it runs successfuly, render with a "Thank you" note
    if request.method == 'POST':
        insert_message(request)
        flash('Thank you for your submission!')
        return redirect(url_for('view'))
    else:
        return render_template('submit.html')

PART II: Viewing Random Submissions

4. random_message(n):

This function should returns to a collection of n random messages from the database

def random_messages(n):
    db_connection = get_message_db()
    # The cursor is used to execute SQL command and process with returned results
    cursor = db_connection.cursor()
    # In this SQL command, 'ORDER BY RANDOM' ensures the result is random, and 'LIMIT' ensures a predefined n number of returned messages
    cmd = 'SELECT handle, message FROM messages ORDER BY RANDOM() LIMIT ?;'
    # fetchall() is to get all matched records
    result = cursor.execute(cmd, (n,)).fetchall()
    return result 
#This function should return a result of a list, each contains handle and message

5. view (): render view.html template:

This function first call random_mesages to grab some random messages and pass these messages as an argument to render_template

def view():
    #The function returns 5 rows of handle and message
    result = random_messages(5)
    # message=results passes the randome messages obtained from the database to the template
    return render_template('view.html',messages=result)

# View Route: Displays a list of messages by fetching a predefined number (e.g., 5) of random messages and rendering them on 'view.html'.    
@app.route('/view')
def view():
    result = random_messages(5)
    return render_template('view.html',messages=result)

# Random Message Route: Similar to the view route, show a varying set of 5 random messages each time the '/random' URL is accessed. 
# Renders 'random.html' with the messages.
@app.route('/random')
def show_random_messages():
    messages = random_messages(5)  # Fetch 5 random messages
    return render_template('random.html', messages=messages)
